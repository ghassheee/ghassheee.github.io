import android.app.Service; import android.content.lntent; import android.os.Binder; import android.util.Log; public class SimpleService extends Service { private final String TAG = SimpleService; //service の 初回起動 時に実行 @Override public void onCreate(){ super.onCreate(); Log.i(TAG, "onCreate"); } @Override public int onStartCommand(Intent intent, int flags, int startId){ Log.i(TAG, "onStartCommand"); return START_STICKY; } //service の bind 時に実行 @Override public IBinder onBind(Intent intent){ return null;
